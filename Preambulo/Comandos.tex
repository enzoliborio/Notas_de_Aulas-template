\newcommand{\fimresolucao}{%
	\ifmmode%
	   \tag*{$\square$}% Modo matemático (alinhado à direita como \tag*)
	\else%
	   \hfill $\square$% Modo texto (alinhado à direita com \hfill)
	\fi%
}

\newcommand{\fimprova}{%
	\ifmmode%
	   \tag*{$\blacksquare$}% Modo matemático
	\else%
	   \hfill $\blacksquare$% Modo texto
	\fi%
}
